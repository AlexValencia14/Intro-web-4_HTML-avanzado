		tr --> crea fila
		th --> columna de encabezado
		td --> crea columna


	form --> Formularios, action="ruta" method="post"

		number
		email
		password
		date
		time
		tel
		file
		radio

	label --> etiquetar elementos

	Tipos de botones <button type="">
		submit --> Enviar
		reset --> Restablecer

	textarea --> Area de texto, cols="50" rows="5"
